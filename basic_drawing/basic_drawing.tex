    \draw[->] (-0.2,0) -- (4.2,0) node[right] {$x$};

        node[right] {$f(x) = \sin x$};
    \draw[color=orange] plot[id=cos] function{cos(x)}
        node[above right] {$f(x) = \cos x$};
